bubble_plot <- function(df) {
  df %>% 
    gather(key = Question, value = Answer) %>% 
    mutate(Question = str_to_lower(Question)) %>%
    ggplot(aes(x = Question, y = Answer)) + 
    geom_count(colour = "#00008B") + 
    theme_minimal() + 
    theme(axis.text.x = element_text(angle = 90, hjust = 1))
}
